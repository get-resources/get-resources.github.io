1.编程求1000以内所有素数

代码如下：

#include<iostream>
#include<cmath>
using namespace std;
int main(){
    int i,n;
    cout<<"1到1000间的素数有："<<endl;
    for(n=2;n<=1000;n++){
        for(i=2;i<=sqrt(n);i++)
            if(n%i==0)
                break;
        if(i>sqrt(n))
            cout<<n<<' ';
    }
    cout<<endl;
    return 0; 
}

2.编写一个函数，将一个十六进制数转换成十进制数，并写一个主函数测试这个函数写的是否正确。（不允许用c++格式控制完成）

代码如下：

#include<iostream>
#include<cstring>
#include<cmath>
using namespace std;
int hex_to_dec(char *a){
    int len = strlen(a);
    int sum = 0;
    for (int i = 0;i < len;i++){
        if(a[i] >= 'A' && a[i] <= 'F')
            a[i] = int(a[i] - 'A') + 10 + '0';
        if(a[i] >= 'a' && a[i] <= 'f')
            a[i] = int (a[i] - 'a') + 10 + '0';
        sum += (a[i] - '0') * (pow(16.0,len - 1 - i));
    }
    return sum;
}
int main(){
    char a[10];
    cout<<"请输入一个十六进制的数："<<endl;
    cin>>a;     
    cout<<"转换成十进制以后："<<hex_to_dec(a)<<endl;
    return 0;
}
